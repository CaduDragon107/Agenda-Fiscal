Estrutura:
```tsx
export default async function TarefaDetalhePage({ params }: { params: Promise<{ id: string }> }) {
  const { id } = await params
  const session = await auth()
  if (!session?.user) redirect("/login")

  const tarefa = await buscarTarefaPorId(session.user, id)
  if (!tarefa) notFound()   // anti-IDOR: tarefa inexistente OU fora do escopo = 404

  const alerta = calcularAlertaPrazo(tarefa.prazo, tarefa.status)
  // ...
}
```

NOTA SOBRE `params`: No Next.js 15 App Router, `params` é uma Promise em componentes assíncronos de rota dinâmica. Usar `await params` para acessar o id. Ver a convenção no projeto existente — verificar se há outros exemplos de `[id]/page.tsx` em `src/app/(app)/empresas/` antes de decidir.

O botão "Marcar como concluída" precisa de estado de loading (pending) ao clicar, então deve ser um Client Component separado. Criar como componente interno `"use client"` no mesmo arquivo via um segundo `export` ou como um arquivo separado `concluir-button.tsx` na mesma pasta.

Implementação recomendada — arquivo separado `src/app/(app)/tarefas/[id]/concluir-button.tsx`:
```tsx
"use client"
import { useState, useTransition } from "react"
import { useRouter } from "next/navigation"
import { Check, Loader2 } from "lucide-react"
import { Button } from "@/components/ui/button"
import { toast } from "sonner"
import { concluirTarefa } from "../actions"

export function ConcluirButton({ tarefaId }: { tarefaId: string }) {
  const [isPending, startTransition] = useTransition()
  const router = useRouter()

  function handleConcluir() {
    startTransition(async () => {
      const result = await concluirTarefa(tarefaId)
      if (!result.ok) {
        toast.error("Não foi possível registrar a conclusão. Tente novamente.")
      } else {
        toast.success("Tarefa marcada como concluída.")
        router.refresh()  // revalida a página de detalhe para ocultar o botão
      }
    })
  }

  return (
    <Button onClick={handleConcluir} disabled={isPending}>
      {isPending
        ? <Loader2 className="animate-spin size-4 mr-2" />
        : <Check className="size-4 mr-2" />}
      Marcar como concluída
    </Button>
  )
}
```

Se preferir manter no mesmo arquivo, extrair como componente local com `"use client"` diretiva própria — não é possível misturar server/client no mesmo arquivo; usar arquivo separado.
